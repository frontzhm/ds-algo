var sumOfLeftLeaves = function(root) {
  if(root === null) return 0
  let sum = 0
  traverse(root.left,true)
  traverse(root.right,false)
  return sum
  function traverse(node,isLeft){
    if(node === null) return
    const isLeafNode = node.left === null && node.right=== null
    if(isLeafNode && isLeft){
      sum+=node.val
      return
    }
    traverse(node.left,true)
    traverse(node.right,false)
  }

};

  1. 在二叉树中增加一行
js
var addOneRow = function(root, val, depth) {
  if(root === null) return null
  if(depth === 1) {
    const newRoot = new TreeNode(val)
    newRoot.left = root
    return newRoot
  }
  const queue = [root]
  let height = 0
  while(queue.length){
    height++
    const curHeightSize = queue.length
    const isCurHeight = depth -1 === height
    for(let i=0;i<curHeightSize;i++){
      const curNode = queue.shift()
      if(isCurHeight){
        const newLeftNode = new TreeNode(val)
        newLeftNode.left = curNode.left
        curNode.left = newLeftNode

        const newRightNode = new TreeNode(val)
        newRightNode.right = curNode.right
        curNode.right = newRightNode
      }else{
        if(curNode.left) queue.push(curNode.left)
        if(curNode.right) queue.push(curNode.right)
      }


    }
    if(isCurHeight) return root
    // 这层遍历完了
  }
  return root


};

  1. 二叉树的堂兄弟节点
js
var isCousins = function(root, x, y) {
  if(root === null) return false
  let res = false
  let xParent = null
  let yParent = null
  let xDepth = 0
  let yDepth = 0
  traverse(root,null,1)
  return xDepth === yDepth && xParent!==yParent

  function traverse(node,parent,depth){
    if(node === null) return
    if(node.val === x){
      xParent = parent
      xDepth = depth
    }
    if(node.val === y){
      yParent = parent
      yDepth = depth
    }
    if(xParent && yParent){return}
    traverse(node.left,node,depth+1)
    traverse(node.right,node,depth+1)
  }

};

var goodNodes = function(root) {
  if(root === null) return 0
  let count = 0
  traverse(root,-Infinity)
  return count
  function traverse(node,maxVal){
    if(node === null) return
    if(node.val >=maxVal){
      maxVal = node.val
      count++
    }
    traverse(node.left,maxVal)
    traverse(node.right,maxVal)

  }

};
