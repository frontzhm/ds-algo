623. 在二叉树中增加一行

```js
var addOneRow = function(root, val, depth) {
  if(root === null) return null
  if(depth === 1) {
    const newRoot = new TreeNode(val)
    newRoot.left = root
    return newRoot
  }
  const queue = [root]
  let height = 0
  while(queue.length){
    height++
    const curHeightSize = queue.length
    const isCurHeight = depth -1 === height
    for(let i=0;i<curHeightSize;i++){
      const curNode = queue.shift()
      if(isCurHeight){
        const newLeftNode = new TreeNode(val)
        newLeftNode.left = curNode.left
        curNode.left = newLeftNode

        const newRightNode = new TreeNode(val)
        newRightNode.right = curNode.right
        curNode.right = newRightNode
      }else{
        if(curNode.left) queue.push(curNode.left)
        if(curNode.right) queue.push(curNode.right)
      }


    }
    if(isCurHeight) return root
    // 这层遍历完了
  }
  return root


};
```

var isCousins = function(root, x, y) {
  if(root === null) return false
  let res = false
  let xParent = null
  let yParent = null
  let xDepth = 0
  let yDepth = 0
  traverse(root,null,1)
  return xDepth === yDepth && xParent!==yParent

  function traverse(node,parent,depth){
    if(node === null) return
    if(node.val === x){
      xParent = parent
      xDepth = depth
    }
    if(node.val === y){
      yParent = parent
      yDepth = depth
    }

  }

};
